Descripción del Proyecto

Proyecto de testing de APIs REST que incluye colección completa de pruebas en Postman para validar endpoints, autenticación, manejo de errores y flujos de trabajo. Implementa automatización con Newman y scripts en Python para integración continua.

Pruebas de Autenticación

JavaScript auth_tests.js 
// Tests para el endpoint de Login
pm.test("Status code es 200", function() {
    pm.response.to.have.status(200);
});

pm.test("Respuesta contiene token", function() {
    var jsonData = pm.response.json();
    pm.expect(jsonData).to.have.property('token');
    pm.expect(jsonData.token).to.not.be.empty;
});

pm.test("Token tiene formato válido JWT", function() {
    var jsonData = pm.response.json();
    var tokenPattern = /^eyJ[A-Za-z0-9-_]+\.[A-Za-z0-9-_]+\.[A-Za-z0-9-_]+$/;
    pm.expect(jsonData.token).to.match(tokenPattern);
});

pm.test("Tiempo de respuesta menor a 500ms", function() {
    pm.expect(pm.response.responseTime).to.be.below(500);
});

pm.test("Content-Type es JSON", function() {
    pm.response.headers.has("Content-Type");
    pm.expect(pm.response.headers.get("Content-Type")).to.include("application/json");
});

// Guardar token para siguientes requests
if (pm.response.code === 200) {
    var jsonData = pm.response.json();
    pm.collectionVariables.set("token", jsonData.token);
    pm.collectionVariables.set("user_id", jsonData.user.id);
}

// Tests para login con credenciales inválidas
pm.test("Login fallido devuelve 401", function() {
    pm.response.to.have.status(401);
});

pm.test("Error contiene mensaje apropiado", function() {
    var jsonData = pm.response.json();
    pm.expect(jsonData.error).to.eql("Credenciales inválidas");
});

// ============ GET ALL ITEMS ============
pm.test("GET - Status 200", function() {
    pm.response.to.have.status(200);
});

pm.test("GET - Array de items presente", function() {
    var jsonData = pm.response.json();
    pm.expect(jsonData).to.have.property('data');
    pm.expect(jsonData.data).to.be.an('array');
});

pm.test("GET - Items tienen estructura válida", function() {
    var jsonData = pm.response.json();
    if (jsonData.data.length > 0) {
        var item = jsonData.data[0];
        pm.expect(item).to.have.all.keys('id', 'name', 'price', 'stock');
        pm.expect(item.id).to.be.a('number');
        pm.expect(item.price).to.be.at.least(0);
    }
});

// ============ CREATE ITEM ============
pm.test("POST - Status 201 Created", function() {
    pm.response.to.have.status(201);
});

pm.test("POST - Item creado con ID", function() {
    var jsonData = pm.response.json();
    pm.expect(jsonData).to.have.property('id');
    pm.expect(jsonData.name).to.eql(pm.variables.get("item_name"));

Automatización con Python

Python test_api_python.py 
import requests
import pytest
from typing import Dict, Any

BASE_URL = "https://api.example.com/v1"

class TestAPI:
    """Suite de pruebas API con Python"""
    
    @pytest.fixture(autouse=True)
    def setup(self):
        """Setup antes de cada prueba"""
        self.base_url = BASE_URL
        self.token = None
        self.item_id = None
    
    def get_auth_token(self) -> str:
        """Obtener token de autenticación"""
        response = requests.post(
            f"{self.base_url}/auth/login",
            json={"email": "test@example.com", "password": "test123"}
        )
        assert response.status_code == 200
        return response.json()["token"]
    
    def create_test_item(self, token: str) -> int:
        """Crear item de prueba y retornar su ID"""
        response = requests.post(
            f"{self.base_url}/items",
            headers={"Authorization": f"Bearer {token}"},
            json={"name": "Test Item", "price": 99.99, "stock": 10}
        )
        assert response.status_code == 201
        return response.json()["id"]
    
    def test_get_items_returns_200(self):
        """Verificar que GET /items retorna 200"""
        response = requests.get(f"{self.base_url}/items")
        assert response.status_code == 200
    
    def test_get_items_returns_array(self):
        """Verificar que la respuesta contiene array de items"""
        response = requests.get(f"{self.base_url}/items")
        data = response.json()
        assert "data" in data
        assert isinstance(data["data"], list)
    
    def test_create_item_with_auth(self):
        """Crear item con autenticación"""
        token = self.get_auth_token()
        response = requests.post(
            f"{self.base_url}/items",
            headers={"Authorization": f"Bearer {token}"},
            json={"name": "New Item", "price": 49.99}
        )
        assert response.status_code == 201
        assert response.json()["name"] == "New Item"
    
    @pytest.mark.parametrize("invalid_price", [-1, -50, -100])
    def test_create_item_rejects_negative_price(self, invalid_price):
        """Verificar que precios negativos son rechazados"""
        token = self.get_auth_token()
        response = requests.post(
            f"{self.base_url}/items",
            headers={"Authorization": f"Bearer {token}"},
            json={"name": "Test", "price": invalid_price}
        )
        assert response.status_code == 400

    def test_unauthorized_access_returns_401(self):
        """Verificar que acceso sin token retorna 401"""
        response = requests.get(f"{self.base_url}/items")
        assert response.status_code == 401
