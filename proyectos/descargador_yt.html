Descargador y extractor de audio de videos YouTube con Flask

Proyecto desarrollado en Python con Flask que permite descargar videos de YouTube y extraer su audio en formato MP3.

    from flask import Flask, render_template, request, send_file
    import pafy
    import moviepy.editor as mp

    app = Flask(__name__)

    @app.route('/envia2', methods=['POST'])
    def envia2():
        url = request.form['url']
        video = pafy.new(url)
        best = video.getbest(preftype="mp4")
        best.download('output/')
        name = 'output/' + video.title + '.mp4'
        clip = mp.VideoFileClip(name)
        clip.audio.write_audiofile('output/' + video.title + '.mp3')
        return send_file('output/' + video.title + '.mp3')
