Descripción del Proyecto

Este proyecto implementa una suite completa de pruebas automatizadas para un sistema de gestión de inventario. La suite incluye pruebas unitarias, pruebas funcionales y pruebas de interfaz gráfica, proporcionando cobertura completa del sistema.

import pytest
import sqlite3
from typing import Tuple

@pytest.fixture
def setup_db() -> Tuple[sqlite3.Connection, sqlite3.Cursor]:
    """Fixture que crea una base de datos en memoria para pruebas"""
    connection = sqlite3.connect(":memory:")
    cursor = connection.cursor()
    cursor.execute("""
        CREATE TABLE articulos (
            codigo INTEGER PRIMARY KEY AUTOINCREMENT,
            nombre TEXT NOT NULL,
            precio REAL NOT NULL,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
    """)
    cursor.execute("""
        INSERT INTO articulos (nombre, precio) 
        VALUES ('Producto de prueba', 50.0)
    """)
    connection.commit()
    yield connection, cursor
    connection.close()

def test_consulta_producto_existente(setup_db):
    """Verifica que se pueda consultar un producto existente"""
    connection, cursor = setup_db
    cursor.execute("SELECT * FROM articulos WHERE nombre = ?", ('Producto de prueba',))
    result = cursor.fetchone()
    assert result is not None
    assert result[2] == 50.0

def test_modificar_producto(setup_db):
    """Verifica la modificación de un producto existente"""
    connection, cursor = setup_db
    nuevo_nombre = 'Producto modificado'
    nuevo_precio = 99.99
    cursor.execute(
        "UPDATE articulos SET nombre = ?, precio = ? WHERE nombre = ?",
        (nuevo_nombre, nuevo_precio, 'Producto de prueba')
    )
    connection.commit()
    
    cursor.execute("SELECT * FROM articulos WHERE nombre = ?", (nuevo_nombre,))
    result = cursor.fetchone()
    assert result is not None
    assert result[2] == nuevo_precio

@pytest.mark.parametrize("precio", [0, -1, -50.5])
def test_precio_invalido_rechazado(setup_db, precio):
    """Verifica que precios inválidos sean rechazados"""
    connection, cursor = setup_db
    with pytest.raises(sqlite3.IntegrityError):
        cursor.execute(
            "INSERT INTO articulos (nombre, precio) VALUES (?, ?)",
            ("Producto", precio)
        )

Pruebas de Integración

Python test_integracion.py 
import unittest
from unittest.mock import patch, MagicMock
import sqlite3

class TestInventarioIntegracion(unittest.TestCase):
    """Pruebas de integración para el flujo completo del sistema"""
    
    def setUp(self):
        """Configura la base de datos de prueba"""
        self.connection = sqlite3.connect(":memory:")
        self.cursor = self.connection.cursor()
        self.cursor.execute("""
            CREATE TABLE articulos (
                codigo INTEGER PRIMARY KEY,
                nombre TEXT,
                precio REAL
            )
        """)
        self.cursor.execute("""
            INSERT INTO articulos (nombre, precio) 
            VALUES ('Producto de prueba', 50.0)
        """)
        self.connection.commit()

    def tearDown(self):
        """Limpia los recursos después de cada prueba"""
        self.connection.close()

    def test_flujo_completo_crud(self):
        """Verifica el flujo completo: Crear, Leer, Actualizar, Eliminar"""
        # CREATE
        self.cursor.execute(
            "INSERT INTO articulos (nombre, precio) VALUES (?, ?)",
            ('Nuevo Producto', 100.0)
        )
        self.connection.commit()
        
        # READ
        self.cursor.execute("SELECT * FROM articulos WHERE nombre = ?", ('Nuevo Producto',))
        producto = self.cursor.fetchone()
        self.assertIsNotNone(producto)
        
        # UPDATE
        self.cursor.execute(
            "UPDATE articulos SET precio = ? WHERE nombre = ?",
            (150.0, 'Nuevo Producto')
        )
        self.connection.commit()
        
        # READ UPDATED
        self.cursor.execute("SELECT precio FROM articulos WHERE nombre = ?", ('Nuevo Producto',))
        precio_actualizado = self.cursor.fetchone()[0]
        self.assertEqual(precio_actualizado, 150.0)
        
        # DELETE
        self.cursor.execute(
            "DELETE FROM articulos WHERE nombre = ?",
            ('Nuevo Producto',)
        )
        self.connection.commit()
        
        # VERIFY DELETED
        self.cursor.execute("SELECT * FROM articulos WHERE nombre = ?", ('Nuevo Producto',))
        self.assertIsNone(self.cursor.fetchone())

Puntos Clave del Testing

  • Aislamiento de pruebas: Cada test usa base de datos en memoria
  • Mocking: Se simulan dependencias externas para pruebas unitarias
  • Fixtures: Reutilización de setup común entre pruebas
  • Parametrización: Pruebas con múltiples valores de entrada
  • Cleanup: Recursos limpiados en tearDown
